create database if not exists Juke;
use Juke;
create table songsdetails(songid int primary key auto_increment,songName varchar(30),singer varchar(30),genere varchar(30),path varchar(40));
create table userdetail(username varchar(30),password varchar(30),email varchar(30),phoneNo long,userid int primary key auto_increment);
create table playlist(playlistid int primary key auto_increment,playListName varchar (30),userid int,foreign key(userid) references userdetail (userid));
create table playlistdetails(playlistid int,foreign key (playlistid) references playlist (playlistid),songid int,foreign key (songid) references songsdetails (songid));
